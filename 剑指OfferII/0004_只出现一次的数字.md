### https://leetcode.cn/problems/WGki4K/
```cpp
class Solution {
public:
    int singleNumber(const vector<int>& nums) {
        map<int, int> mp1;
        for(int i=0; i<nums.size(); i++) {
            if(mp1.find(nums[i]) != mp1.end()) {
                mp1[nums[i]] += 1;
            } else {
                mp1[nums[i]] = 1;
            }
        }
        for(const auto &p : mp1) {
            if(p.second == 1) {
                return p.first;
            }
        }
        return -1;
    }
};
```
