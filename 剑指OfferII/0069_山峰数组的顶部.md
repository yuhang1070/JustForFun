### https://leetcode.cn/problems/B1IidL/
```java
class Solution {
    public int peakIndexInMountainArray(int[] arr) {
        int len1 = arr.length - 1;
        for(int i = 1; i < len1; i++) {
            if(arr[i] > arr[i - 1] && arr[i] > arr[i + 1]) {
                return i;
            }
        }
        return 1;
    }
}
```
