class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param n int整型 
     * @param k int整型 
     * @return int整型
     */
    int numsOfStrings(int n, int k) {
        // write code here
        vector<vector<int>> dp(n + 1, vector<int>(k + 1, 0));
        dp[0][0] = 1;
        dp[1][0] = 1;
        dp[2][1] = 26;
        if(n > 2) dp[3][1] = 26;
        for(int i = 4; i <= n; ++i){
            int tk = n >> 1;
            tk = min(k, tk);
            for(int j = 1; j <= tk; ++j){
                dp[i][j] = dp[i - 1][j] + 25 * dp[i - 2][j - 1];
                dp[i][j] %= 1000000;
            }
        }
        return dp[n][k];
    }
};
