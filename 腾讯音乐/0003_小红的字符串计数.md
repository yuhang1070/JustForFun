class Solution {
public:
    /**

动态规划，dp[i][j]表示长度为i，可以分为j段的字符串个数。

首先考虑动态转移方程，假设向一个字符串s添加字符c，如果c与s的最后一个字符相同, 转移方程dp[i][j] += dp[i-1][j]，由i-1长度, 段数为j的字符串转移而来；否则,dp[i][j] += 25*dp[i-2][j-1], 由长度i-2，段数为j-1的字符串转移而来，且当前选择的字符c有25中选择。所以dp[i][j] = dp[i - 1][j] + 25*dp[i - 2][j - 1]。

     */
    int numsOfStrings(int n, int k) {
    const int mod = 1e6;
    // write code here
    int m = min(k, n / 2);
    vector> dp(n + 1, vector(k + 1, 0));
    // 初始化
    for (int i = 2; i <= n; ++i) dp[i][1] = 26;
    for (int i = 4; i <= n; ++i) {
        for (int j = 2; j <= m; ++j) {
            dp[i][j] = dp[i - 1][j] + 25*dp[i - 2][j - 1];
            dp[i][j] %= mod;
        }
    }
    return dp[n][k];
}
};
