            int tk = n >> 1;
            tk = min(k, tk);
            for(int j = 1; j <= tk; ++j){

            }
        }
        return dp[n][k];
    }
};

    const int mod = 1e6;

    for (int i = 2; i <= n; ++i) dp[i][1] = 26;
