#include <iostream>
#include <string>
#include <map>

using namespace std;

#define ULL unsigned long long

map<pair<int, int>, ULL> cache;

ULL func(int m, int n) {
    auto pr = make_pair(m, n);
    auto it = cache.find(pr);
    if (it != cache.end()) {
        return it->second;
    }
    if (!m) {
        return cache[pr] = n;
    } else if (!n) {
        return cache[pr] = m;
    } else {
        return cache[pr] = func(m - 1, n) + func(m, n - 1) + 2;
    }
}

int main() {
    ULL k;
    int n, m;
    cin >> n >> m >> k;
    string cur = "a";
    n--;
    k--;
    while (k > 0 && (m || n)) {
        ULL step = func(n, m) + 1;//子树的个数
        if (step > k) { //k在子树中
            k--;
            if (n) {
                cur += "a";
                n--;
            } else {
                cur += "b";
                m--;
            }
        } else {//k不在子树中，在下一个子树里
            k -= step;
            n++;
            m--;
            cur.back()++;
        }
    }
    cout << cur << endl;
    return 0;
}
