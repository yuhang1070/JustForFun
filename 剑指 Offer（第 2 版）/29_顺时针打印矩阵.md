## 29_顺时针打印矩阵
### https://leetcode.cn/problems/shun-shi-zhen-da-yin-ju-zhen-lcof/
```java
class Solution {
    public int[] spiralOrder(int[][] matrix) {
        if(matrix == null) {
            return new int[0];
        }
        int m = matrix.length;
        if(m == 0) {
            return new int[0];
        }
        int n = matrix[0].length;
        if(n == 0) {
            return new int[0];
        }
        int[] ans = new int[m * n];
        int ix = 0;
        int x0 = 0, y0 = 0;
        int x1 = m - 1, y1 = n - 1;
        while(x0 <= x1 && y0 <= y1) {
            for(int i = y0; i <= y1; i++) {
                ans[ix++] = matrix[x0][i];
            }
            if(ix >= ans.length) {
                break;
            }
            for(int i = x0 + 1; i <= x1 - 1; i++) {
                ans[ix++] = matrix[i][y1];
            }
            if(ix >= ans.length) {
                break;
            }
            for(int i = y1; i >= y0; i--) {
                ans[ix++] = matrix[x1][i];
            }
            if(ix >= ans.length) {
                break;
            }
            for(int i = x1 - 1; i >= x0 + 1; i--) {
                ans[ix++] = matrix[i][y0];
            }
            x0++;
            x1--;
            y0++;
            y1--;
        }
        return ans;
    }
}
```
