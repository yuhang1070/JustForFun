## 剪绳子 II
### https://leetcode.cn/problems/jian-sheng-zi-ii-lcof/
```java
class Solution {
    private int MOD = 1000000007;
    public int cuttingRope(int n) {
        if(n == 2) {
            return 1;
        } else if(n == 3) {
            return 2;
        } else if(n == 4) {
            return 4;
        }
        int x = n / 3;
        int y = n - 3 * x;
        if(y == 0) {
            return (int) myPow(x);
        } else if(y == 1) {
            return (int)(myPow(x - 1) * 4 % MOD);
        } else {
            return (int) (myPow(x) * 2 % MOD);
        }
    }
    private long myPow(int x) {
        long ans = 1;
        for(int i = 0; i < x; i++) {
            ans = ans * 3 % MOD;
        }
        return ans;
    }
}
```
