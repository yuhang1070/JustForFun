## 矩阵中的路径
### https://leetcode.cn/problems/ju-zhen-zhong-de-lu-jing-lcof/
```java
class Solution {
    public boolean exist(char[][] board, String word) {
        int m = board.length;
        int n = board[0].length;
        if(m * n < word.length()) {
            return false;
        }
        boolean[][] fs = new boolean[m][n];
        char ch0 = word.charAt(0);
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(ch0 == board[i][j]) {
                    fs[i][j] = true;
                    if(dfs(board, i, j, word, 1, fs)) {
                        return true;
                    }
                    fs[i][j] = false;
                }
            }
        }
        return false;
    }
    public boolean dfs(char[][] board, int i, int j, String word, int cnt, boolean[][] fs) {
        int m = board.length;
        int n = board[0].length;
        if(cnt == word.length())
            return true;
        char now = word.charAt(cnt);
        if(i > 0 && board[i - 1][j] == now && !fs[i - 1][j]) {
            fs[i - 1][j] = true;
            if(dfs(board, i - 1, j, word, cnt + 1, fs)) {
                fs[i - 1][j] = false;
                return true;
            }
            fs[i - 1][j] = false;
        }
        if(j > 0 && board[i][j - 1] == now && !fs[i][j - 1]) {
            fs[i][j - 1] = true;
            if(dfs(board, i, j - 1, word, cnt + 1, fs)) {
                fs[i][j - 1] = false;
                return true;
            }
            fs[i][j - 1] = false;
        }
        if(i < m - 1 && board[i + 1][j] == now && !fs[i + 1][j]) {
            fs[i + 1][j] = true;
            if(dfs(board, i + 1, j, word, cnt + 1, fs)) {
                fs[i + 1][j] = false;
                return true;
            }
            fs[i + 1][j] = false;
        }
        if(j < n - 1 && board[i][j + 1] == now && !fs[i][j + 1]) {
            fs[i][j + 1] = true;
            if(dfs(board, i, j + 1, word, cnt + 1, fs)) {
                fs[i][j + 1] = false;
                return true;
            }
            fs[i][j + 1] = false;
        }
        return false;
    }
}
```
