## 小美的书架
### https://www.nowcoder.com/questionTerminal/b34000b87e92468d86f380dc02c97eb9
```java
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int M = sc.nextInt();
        int Q = sc.nextInt();
        BookManager bm = new BookManager(M, N);
        for (int i = 0; i < Q; i++) {
            int o = sc.nextInt();
            if (o == 1) {
                int x = sc.nextInt();
                int y = sc.nextInt();
                bm.op1(x, y);
                continue;
            }
            int x = sc.nextInt();
            if (o == 2) {
                bm.op2(x);
            } else if (o == 3) {
                bm.op3(x);
            } else if (o == 4) {
                bm.op4(x);
            } else if (o == 5) {
                bm.op5(x);
            }
        }
    }
}

class BookManager {
    // 1-N是否上锁
    public HashMap<Integer, Boolean> fs = new HashMap<>();
    // 小美有哪些书
    public HashMap<Integer, Boolean> ms = new HashMap<>();
    // 书号 行号
    public HashMap<Integer, Integer> hs = new HashMap<>();
    public int N;
    public int M;

    public BookManager(int N, int M) {
        this.N = N;
        this.M = M;
        for (int i = 1; i <= N; i++) {
            fs.put(i, false);
        }
        for (int i = 1; i <= M; i++) {
            ms.put(i, true);
        }
        for (int i = 1; i <= M; i++) {
            hs.put(i, 0);
        }
    }

    public void op1(int x, int y) {
        if (x > M || x < 1) {
            return;
        }
        if (y > N || y < 1) {
            return;
        }
        if (!ms.get(x)) {
            return;
        }
        if (fs.get(y)) {
            return;
        }
        if (hs.get(x) != 0) {
            int h = hs.get(x);
            if (fs.get(h)) {
                return;
            }
            hs.put(x, y);
        } else {
            hs.put(x, y);
        }
    }

    public void op2(int y) {
        if (y > N || y < 1) {
            return;
        }
        fs.put(y, true);
    }

    public void op3(int y) {
        if (y > N || y < 1) {
            return;
        }
        fs.put(y, false);
    }

    public void op4(int x) {
        if (x > M || x < 1) {
            System.out.println(-1);
            return;
        }
        if (!ms.get(x)) {
            System.out.println(-1);
            return;
        }
        if (hs.get(x) == 0) {
            System.out.println(-1);
            return;
        }
        int h = hs.get(x);
        if (fs.get(h)) {
            System.out.println(-1);
            return;
        }
        System.out.println(h);
        hs.put(x, 0);
        ms.put(x, false);
    }

    public void op5(int x) {
        if (x > M || x < 1) {
            return;
        }
        ms.put(x, true);
    }
}
```
