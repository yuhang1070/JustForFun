// 超时做法
import java.util.*;
public class Main {
    public static void solve(Scanner sc) {
        int N = sc.nextInt();
        char[] arr1 = sc.next().toCharArray();
        int M = sc.nextInt();
        char[] arr2 = sc.next().toCharArray();
        int minI = 0;
        while(arr1[minI] == '2') {
            minI++;
        }
        int mi = minI, fi = minI;
        boolean has1 = true;
        boolean has0 = true;
        for(int i = 0; i < M; i++) {
            while(arr1[minI] == '2') {
                minI++;
            }
            if(arr2[i] == 'M') {
                if(has1) {
                    boolean find1 = false;
                    for(; mi < N; mi++) {
                        if(arr1[mi] == '1') {
                            find1 = true;
                            break;
                        }
                    }
                    if(find1) {
                        arr1[mi] = '2';
                        System.out.println(mi + 1);
                    } else {
                        mi = minI;
                        for(; mi < N; mi++) {
                            if(arr1[mi] == '0') {
                                break;
                            }
                        }
                        arr1[mi] = '1';
                        System.out.println(mi + 1);
                        has1 = true;
                    }
                } else {
                    for(; mi < N; mi++) {
                        if(arr1[mi] == '0') {
                            break;
                        }
                    }
                    arr1[mi] = '1';
                    System.out.println(mi + 1);
                    has1 = true;
                }
            } else {
                if(has0) {
                    boolean find0 = false;
                    for(; fi < N; fi++) {
                        if(arr1[fi] == '0') {
                            find0 = true;
                            break;
                        }
                    }
                    if(find0) {
                        arr1[fi] = '1';
                        System.out.println(fi + 1);
                        mi = Math.min(mi, fi);
                    } else {
                        fi = minI;
                        for(; fi < N; fi++) {
                            if(arr1[fi] == '1') {
                                break;
                            }
                        }
                        arr1[fi] = '2';
                        System.out.println(fi + 1);
                        has0 = false;
                    }
                } else {
                    for(; fi < N; fi++) {
                        if(arr1[fi] == '1') {
                            break;
                        }
                    }
                    arr1[fi] = '2';
                    System.out.println(fi + 1);
                }
            }
        }
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        for(int i = 0; i < T; i++) {
            solve(sc);
        }
    }
}
